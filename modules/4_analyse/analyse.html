#' Simulated dataset
#'
#' @param n number of observations
#' @param beta_trt,beta_bm1,beta_bm2 the true coefficients
#' @param eos non-informative censoring (end of study)
#' @param seed RNG seed
surv_example = function(n=300, 
                  beta_trt=log(0.7), beta_bm1=log(1.5), beta_bm2=log(0.8),
                  eos=30, seed=42){
  set.seed(seed)
  # Generate covariates
  baseline = tibble(
    subjid = seq(n),
    arm = rep(c(0,1), each=n/2),   
    bm1 = rbinom(n, 1, 0.4),
    bm2 = rnorm(n),
  )

  rtn = baseline %>%
    mutate(
      # Generate exponential survival time
      log_predlin = beta_trt*arm + beta_bm1*bm1 + beta_bm2*bm2,
      lambda0 = log(2)/10 / exp(beta_bm1*mean(bm1) + beta_bm2*mean(bm2)),
      lambda = lambda0 * exp(log_predlin),
      t_real = rexp(n, rate=lambda),

      # Administrative censorship
      t = pmin(t_real, eos),
      event = as.numeric(t_real<=eos),

      # labels
      arm =  factor(arm,  labels=c("Control", "Treatment")),
      bm1 = factor(bm1, labels=c("Negative", "Positive")),
    )

  rtn %>%
    select(subjid, arm, t, event, bm1, bm2)
}
